<data_type> ->  number

<state>     ->  return <retun>

<state>     ->  local id <id_n> : <types> = <declr_types>

<id_n>      ->  eps

<func_param_n>  -> , <expression> <func_param_n>

<else>      ->  eps

<init_func> ->  <state>
<init_func> ->  function id ( <params> )

<init_func> ->  <id_n>
